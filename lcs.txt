#include <stdio.h>
int c[20][20];
char b[20][20];
void Lcs_length(char x[],char y[],int m,int n )
{
for(int i=0;i<m;i++)
{
c[i][0]=0;
}
for(int j=0;j<n;j++)
{
c[0][j]=0;
}
for(int i=1;i<=m;i++) { 
    for(int j=1;j<=n;j++)
{
if(x[i-1]==y[j-1])
{
c[i][j]=c[i-1][j-1]+1;
b[i][j]='d';
}
else if(c[i-1][j]>=c[i][j-1]){
c[i][j]=c[i-1][j];
b[i][j]='u';
}
else{
c[i][j]=c[i][j-1];
b[i][j]='l';
}
}
}
}
void print_lcs(char b[20][20],char x[],int i,int j)
{
// printf(“%d %d: %c\n”,i,j,b[i][j]);
if(i==0||j==0){
return;
}
if(b[i][j]=='d'){
// printf(“d: %c\n”,b[i][j]);
print_lcs(b,x,i-1,j-1);
printf("%c",x[i-1]);
}
else if(b[i][j]=='u')
{
// printf(“u: %c\n”,b[i][j]);
print_lcs(b,x,i-1,j);
}
else{
// printf(“l: %c\n”,b[i][j]);
print_lcs(b,x,i,j-1); }
}
